import java.util.Scanner;

public class TempProgram {

	public static void main(String[] args) {

		Scanner input = new Scanner(System.in);
		Product[] products = new Product[5];

		for (int i = 0; i < products.length; i++) {

			// Handle product name prompt and input
			System.out.print("Product " + (i + 1) + " name\t: ");
			String name = input.nextLine();

			// Handle price prompt and input validation
			double price;
			while (true) {
				System.out.print("Product " + (i + 1) + " price\t: ");
				if (input.hasNextDouble()) {
					price = input.nextDouble();
					if (price > 0) {
						break;
					} else {
						System.out.println("Please enter a value above 0.");
					}
				} else {
					System.out.println("Please enter a valid price.");
					input.nextLine();
				}
			}

			input.nextLine();

			// Handle quantity prompt and input validation
			int quantity;
			while (true) {
				System.out.print("Product " + (i + 1) + " qty.\t: ");
				if (input.hasNextInt()) {
					quantity = input.nextInt();
					if (quantity >= 0) {
						break;
					} else {
						System.out.println("Please enter a number greater than or equal to 0.");
					}
				} else {
					System.out.println("Please enter a number.");
					input.nextLine();
				}
			}

			input.nextLine();
			System.out.println();
			products[i] = new Product(name, price, quantity);

			if (i < products.length - 1) {
				while (true) {
					System.out.print("Continue to the next product? (y/n): ");
					switch (input.nextLine()) {
						case "y":
						System.out.println();
						break;
						case "n":
						printProducts(products, input);
						return;
						default:
						System.out.println("Invalid input (y/n).");
						input.nextLine();
					}
					break;
				}
			}
		}

		printProducts(products, input);
		input.close();
	}

	public static void printProducts(Product[] products, Scanner input) {
		while (true) {
			System.out.print("Print details? (y/n) : ");
			switch (input.nextLine()) {
				case "y":
				System.out.println();
				for (Product product: products) {
					if (product != null) {
						product.cntrtPrint();
					}
				}
				break;
				case "n":
				break;
				default:
				System.out.println("Invalid input (y/n).");
				input.nextLine();
			}

			break;
		}
	}
}

class Product {

	String name;
	double price;
	int quantity;

	Product(String name, double price, int quantity) {

		this.name = name;
		this.price = price;
		this.quantity = quantity;
	}

	public void cntrtPrint() {

		System.out.println("Item name\t: " + name);
		System.out.println("Item price\t: " + "\u20B1 " + price);
		System.out.println("Item qty\t: " + quantity + " pcs.");
		System.out.println();
	}
}
