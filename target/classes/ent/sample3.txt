		System.out.print("Please type the student name\t: ");

	public static double getValidInput(Scanner input) {
		// getValidInput, variant two, revision two

		double value = 0;
		String message = "Please type the student grade";
		String fdbkOne = "Invalid input. Please enter a number.";
		String fdbkTwo = "Please enter a mark within 70 - 100.";

		while (true) {
			try {
				System.out.print(message + "\t: ");
				value = input.nextDouble();

				if (value >= 70 && value <= 100) {
					break;
				} else {
					System.out.println(fdbkTwo);
				}

			} catch (java.util.InputMismatchException e) {
				System.out.println(fdbkOne);
				input.nextLine();
			}
		}

		return value;
	}
